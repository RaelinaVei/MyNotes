// --- UPDATED SYNCED TIMER LOGIC --- let timerId = null; function updateDisplay(seconds) { const m = Math.floor(seconds / 60); const s = seconds % 60; const timeStr = `${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`; document.getElementById('timerDisplay').textContent = timeStr; } function toggleTimer() { const isRunning = localStorage.getItem('timerRunning') === 'true'; const btn = document.getElementById('pomoBtn'); if (!isRunning) { // STARTING const mins = document.getElementById('studyMin').value; const expiryTime = Date.now() + (mins * 60 * 1000); localStorage.setItem('timerExpiry', expiryTime); localStorage.setItem('timerRunning', 'true'); btn.textContent = "PAUSE SESSION"; startInterval(); } else { // PAUSING clearInterval(timerId); localStorage.setItem('timerRunning', 'false'); btn.textContent = "RESUME SESSION"; } } function startInterval() { if(timerId) clearInterval(timerId); timerId = setInterval(() => { const expiry = localStorage.getItem('timerExpiry'); const now = Date.now(); const diff = Math.max(0, Math.floor((expiry - now) / 1000)); updateDisplay(diff); if (diff <= 0) { clearInterval(timerId); localStorage.setItem('timerRunning', 'false'); alert("Time is up!"); } }, 500); } function resetTimer() { clearInterval(timerId); localStorage.setItem('timerRunning', 'false'); localStorage.removeItem('timerExpiry'); const mins = document.getElementById('studyMin').value; updateDisplay(mins * 60); document.getElementById('pomoBtn').textContent = "START SESSION"; } // Check on load if timer was already running from elsewhere window.onload = () => { if(localStorage.getItem('timerRunning') === 'true') { startInterval(); document.getElementById('pomoBtn').textContent = "PAUSE SESSION"; } };
